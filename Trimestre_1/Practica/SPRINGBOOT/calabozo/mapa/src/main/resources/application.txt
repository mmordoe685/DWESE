spring.application.name=mapa
server.port=8080

#Parametros de conexion a la bd Mysql
spring.datasource.url=jdbc:mysql://localhost:3306/citydungeon 
spring.datasource.username=root
spring.datasource.password=toor

#Este comando hace que hibernate automaticamente cree las tablas de las clases @Entity
spring.jpa.hibernate.ddl-auto=update
#Para que se muestre por consola las sentencias sql
spring.jpa.show-sql=true

spring.jpa.database-platform=org.hibernate.dialect.MySQLDialect
spring.mvc.format.date=yyyy-MM-dd

# Spring Boot OAuth2 Configuration
spring.security.oauth2.client.registration.google.client-id=
spring.security.oauth2.client.registration.google.client-secret=
spring.security.oauth2.client.registration.google.scope=openid,profile,email
spring.security.oauth2.client.registration.google.redirect-uri=http://localhost:8080/login/oauth2/code/google

# Configuración específica para Google OIDC
spring.security.oauth2.client.provider.google.issuer-uri=https://accounts.google.com

# Logging (opcional)
logging.level.org.springframework.security=DEBUG
logging.level.org.springframework.web=DEBUG
logging.level.org.springframework.web.client.RestTemplate=DEBUG
logging.level.org.springframework.security.oauth2=TRACE

# Nivel de log general de la aplicación
logging.level.root=INFO

# Archivo donde se guardarán los logs
logging.file.name=logs/aplicacion.log

# Tamaño máximo del archivo de log
logging.logback.rollingpolicy.max-file-size=10MB

# Formato de salida en consola
logging.pattern.console=%d{yyyy-MM-dd HH:mm:ss} - %msg%n
